273  static std::vector<std::string> split(const std::string &str, char splitor,
274  bool ignoreEmpty = true) {
275  std::vector<std::string> vec;
276  auto iter = str.cbegin();
277  while (true) {
278  auto pos = std::find(iter, str.cend(), splitor);
279  auto elem = std::string(iter, pos);
280  if ((!ignoreEmpty) || (ignoreEmpty && !elem.empty()))
281  vec.push_back(elem);
282  if (pos == str.cend())
283  break;
284  iter = ++pos;
285  }
286  return vec;
287  }
